#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

int n,m;
vector<int> arr;
vector<int> make;
vector<bool> visit;
vector<vector<int>> ans;

void dfs(int at, int depth){
    if(depth == m){
        ans.push_back(make);
        return;
    }
    for(int i = 0; i<n ;i++){
        if(visit[i] == false) {
            make[depth] = arr[i];
            visit[i] = true;
        }
        else
            continue;
        dfs(i+1, depth + 1);
        visit[i] = false;
    }
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    cin >> n >> m;

    for(int i=0; i<n; i++){
        int temp;   cin >> temp;
        arr.push_back(temp);
        make.push_back(0);
        visit.push_back(false);
    }
    sort(arr.begin(), arr.end());
    dfs(0,0);
    for(int i=0; i<ans.size(); i++){
        for(int j=0; j<m; j++){
            cout << ans[i][j] << ' ';
        }
        cout << "\n";
    }
}
